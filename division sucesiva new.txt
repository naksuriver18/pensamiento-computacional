#include <iostream>
#include <vector>
using namespace std;
int divisionSucesiva(int n, int base, vector<int>&arr){
    int temp = 0;

    while(n !=0){
        arr.push_back(n % base);
        n/= base;
    }
    for(int i = arr.size() - 1; i >= 0; i--){
        temp = temp * 10 + arr[i];
    }
    return temp;
}
int main(){
    vector<int>arr;
    cout<<divisionSucesiva(156, 8, arr);
}
